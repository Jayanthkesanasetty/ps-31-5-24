1. Write a program to print true if the number is unique else print false if it contains duplicates

        -------code-------

        a=prompt("enter number: ")
        b=[]
        for(i=0;i< a.length;i++){
        for(j=i+1;j< a.length;j++){
            if(a[i]==a[j] &&  !b.includes(a[i])){
                b.push(a[i])
            }
            }
            }
        if(b.length==0){
            console.log("NO DUPS")
            }
        else{
        console.log("true")
        }
